package main

import (
    "fmt"
    "time"
)

func main() {
    // Get the current time
    now := time.Now()

    // Format the time in RFC3339 format
    formattedTime := now.Format(time.RFC3339)

    // Print the formatted time
    fmt.Println("Current time in RFC3339 format:", formattedTime)
}
